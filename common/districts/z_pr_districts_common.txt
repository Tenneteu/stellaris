	base_buildtime = 480
	is_capped_by_modifier = no

	}
	show_on_uncolonized = {
		always = no	#|z暂时|
		exists = from
		uses_district_industrial = yes
		from = { has_country_flag = z_pr_flg_cntr_reg }
	}
	potential = {
		always = no	#|z暂时|
		exists = owner
		uses_district_industrial = yes
		owner = { has_country_flag = z_pr_flg_cntr_reg }
	}
	allow = {
		NOT = { has_modifier = resort_colony }
		NOT = { has_modifier = slave_colony }
	}

	conversion_ratio = 1
	convert_to = {
		district_industrial
		district_arcology_arms_industry
	#	z_pr_dstr_forge
		z_pr_dstr_industrial
	}

	
	resources = {
		category = planet_districts_industrial
		cost = {
			minerals = @city_cost
		}
		upkeep = {
			energy = 2
		}
		upkeep = {
			trigger = {
				exists = owner
				owner = {
					has_edict = industrial_maintenance
				}
			}
			energy = 2
		}
	}

	planet_modifier = {
		planet_housing_add = 3
	}


	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_regular_empire = yes }
		}
		job_foundry_add = 3
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_hive_empire = yes }
		}
		job_alloy_drone_add = 3
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_machine_empire = yes }
		}
		job_fabricator_add = 3
	}
	#_[专精]
	triggered_planet_modifier = {
		potential = {
			has_designation = col_foundry
			exists = owner
			owner = { is_regular_empire = yes }
		}
		modifier = {
			planet_housing_add = 1
			job_foundry_add = 1
		}
	}
	triggered_planet_modifier = {
		potential = {
			has_designation = col_foundry
			exists = owner
			owner = { is_hive_empire = yes }
		}
		modifier = {
			planet_housing_add = 1
			job_alloy_drone_add = 1
		}
	}
	triggered_planet_modifier = {
		potential = {
			OR = {
				has_designation = col_factory
				has_designation = col_mac_factory
			}
			exists = owner
			owner = { is_machine_empire = yes }
		}
		modifier = {
			planet_housing_add = 1
			job_fabricator_add = 1
		}
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_regular_empire = yes }
		}
		text = job_foundry_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_hive_empire = yes }
		}
		text = job_alloy_drone_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_machine_empire = yes }
		}
		text = job_fabricator_effect_desc
	}

	triggered_planet_modifier = {

		}

			exists = owner
			owner = {
				is_machine_empire = yes
				has_edict = industrial_maintenance
			}
		}
		text = job_maintenance_drone_effect_desc
	}

		trigger = { always = yes }
