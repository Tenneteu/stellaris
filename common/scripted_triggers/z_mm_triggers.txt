###_[Hidden_buttons]	##############################################################################
#_[country]	拓展UI
z_mm_trgr_expanded_sub_ui = {
	OR = {
		AND = { is_scope_type = country		has_country_flag = z_mm_flg_expanded_sub_ui_on }
		AND = { exists = owner		owner = { has_country_flag = z_mm_flg_expanded_sub_ui_on } }
	}
}

	}
